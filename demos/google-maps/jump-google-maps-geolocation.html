Using jump.js with Google maps

Download jQuery 1.4.X or higher or use Googles or Microsofts CDN. 

<script type="text/javascript" src="http://maps.google.com/maps/api/js?sensor=false"></script>
<script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min.js"></script>
<script type="text/javascript" src="PATH_TO_PLUGIN/jump.js"></script>
<script type="text/javascript" src="PATH_TO_PLUGIN/jump.extensions.js"></script>
$('#map_canvas').jump().bind('init', function(evt, map) {
	$('#map_canvas').jump('getCurrentPosition', function(result, status) {
		if ( status === 'OK' ) {
			var lat = result.position.coords.latitude;
			var lng = result.position.coords.longitude;
			var position = new google.maps.LatLng(lat, lng);
			$('#map_canvas').jump('addMarker', {'position': position, 'bounds': true});
		}
	});   
});
$('#map_canvas').jump({}, function() {
	var self = this;
	self.getCurrentPosition(function(result, status) {
		if ( status === 'OK' ) {
			var lat = result.position.coords.latitude;
			var lng = result.position.coords.longitude;
			var position = new google.maps.LatLng(lat, lng);
			self.addMarker({ 'position': position, 'bounds': true });
		}
	});   
});
