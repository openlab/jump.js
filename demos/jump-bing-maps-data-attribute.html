Using jump.js with Bing maps

Download jQuery 1.4.X or higher or use Googles or Microsofts CDN. 

<script type="text/javascript" src="http://ecn.dev.virtualearth.net/mapcontrol/mapcontrol.ashx?v=7.0"></script>
<script type="text/javascript" src="http://ajax.aspnetcdn.com/ajax/jQuery/jquery-1.7.2.min.js"></script>
<script type="text/javascript" src="PATH_TO_PLUGIN/jump.js"></script>
$('#map_canvas').jump('credentials': YOUR_BING_MAP_CREDENTIALS).bind('init', function(ev, map) {
	$("[data-gmapping]").each(function(i, el) {
		var data = $(el).data('gmapping');
		$('#map_canvas').jump('addMarker', {
				'id': data.id, 
				'tags': data.tags,
				'title': data.title
				'location': new Microsoft.Maps.Location(data.latlng.lat, data.latlng.lng), 
				'bounds': true 
			}, function(map, marker) {
				$(el).click(function() {
					$(marker).triggerEvent('click');
				}).hover(function() {
					map.setView({ 'center': marker.getLocation() });
				});
			}	
		).click(function() {
			$('#map_canvas').jump('openInfoWindow', { 
				'title': this.target.title, 
				'description': 'Static description' 
			}, this);
		});
	});	
});
